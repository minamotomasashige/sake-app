    function processImg(input, previewId) {
        if (!input.files[0]) return;
        const reader = new FileReader();
        reader.onload = (e) => {
            const img = new Image();
            img.src = e.target.result;
            img.onload = () => {
                const canvas = document.createElement('canvas');
                const MAX = 800;
                let w = img.width, h = img.height;
                if (w > h) { if (w > MAX) { h *= MAX / w; w = MAX; } }
                else { if (h > MAX) { w *= MAX / h; h = MAX; } }
                canvas.width = w; canvas.height = h;
                canvas.getContext('2d').drawImage(img, 0, 0, w, h);
                curImg = canvas.toDataURL('image/jpeg', 0.8);
                document.getElementById(previewId).innerHTML = `<img src="${curImg}" style="width:100px; border-radius:8px;">`;
            };
        };
        reader.readAsDataURL(input.files[0]);
    }

    function runSearch() {
        const kw = document.getElementById('s-kw').value.toLowerCase();

        const limOnly = document.getElementById('f-lim-only').checked;
        const sort = document.getElementById('s-sort').value;

            const matchText = (s.name + s.brewery + s.pref + s.memo + s.tags.join('')).toLowerCase().includes(kw);

            if (akOnly) matchFilter = (s.no >= 1 && s.no <= 30) || (s.no >= 51 && s.no <= 100);

            return matchText && matchFilter;

        else res.sort((a, b) => b.id - a.id);

    function renderShelf() {
        const draw = (start, end, containerId) => {
            const cont = document.getElementById(containerId);
            cont.innerHTML = "";
            for(let i = start; i <= end; i++) {
                if (i >= 31 && i <= 49) continue; 
                const s = sakeDB.find(x => x.no === i);
                cont.innerHTML += `
                    <div class="shelf-cell" onclick="${s ? `openSakeDetail(${s.id})` : ''}">
                        <div class="shelf-no">${i}</div>
                        ${s ? `<img src="${s.img}" style="width:100%; height:100%; object-fit:cover;">` : ''}
                    </div>`;
            }
        };
        draw(1, 30, 'shelf-1-30');
        draw(51, 100, 'shelf-51-100');
    }

    function renderHistory() {
        const cont = document.getElementById('history-list');
        cont.innerHTML = "";
        const prefs = [...new Set(sakeDB.map(s => s.pref))].filter(x => x);
        prefs.forEach(p => {
            const list = sakeDB.filter(s => s.pref === p);
            cont.innerHTML += `<div style="margin-bottom:15px;"><div style="background:#eee; padding:8px 15px; font-weight:bold;">${p}</div><div class="horizontal-scroll" style="padding:10px;">${list.map(s => `<div class="sake-mini-card" style="flex:0 0 90px;" onclick="openSakeDetail(${s.id})"><img src="${s.img}"><div style="padding:5px;">${s.name}</div></div>`).join('')}</div></div>`;
        });
    }

    }

    async function deleteItem(key, id) {
        if(!confirm("å‰Šé™¤ã—ã¾ã™ã‹ï¼Ÿ")) return;
        if(key === 'sakes') sakeDB = sakeDB.filter(s => s.id !== id);
        else breweryDB = breweryDB.filter(b => b.id !== id);
        await localforage.setItem(key, key === 'sakes' ? sakeDB : breweryDB);
        closeModal();
        location.reload();
    }
